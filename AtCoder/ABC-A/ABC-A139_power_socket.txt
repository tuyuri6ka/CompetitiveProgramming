process.stdin.resume();
process.stdin.setEncoding('utf8');

function main(input) {
	const args = input.trim().split("\n").map(e => e.split(' ').map(e => parseInt(e, 10)));
	const [A, B] = args.shift();

	let plug = 1;
	let tap  = 0;
	while (plug < B) {
		tap++;
		plug = 1 + tap * (A - 1);
	}

	console.log(tap);
}

main(require("fs").readFileSync("/dev/stdin", "utf8"));
